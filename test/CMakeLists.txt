macro(compile_test arg)
  add_executable(${arg} ${arg}.cpp)
  target_link_libraries(${arg} mcsapi ${LIBUV_LIBRARY} ${LIBXML2_LIBRARIES} ${GTEST_BOTH_LIBRARIES} ${MYSQL_LIBRARY} ${CMAKE_THREAD_LIBS_INIT})
  add_test(${arg} ${arg})
endmacro(compile_test)
